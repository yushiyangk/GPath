 59	def __init__(self, path: Union[PathLike, GPath, None]="") -> None:
 60		"""
 61			Initialise a normalised and generalised abstract file path, possibly by copying an existing GPath object.
 62
 63			Parameters
 64			----------
 65			`path`
 66			: path-like object representing a (possibly unnormalised) file path, or a GPath object to be copied
 67
 68			Raises
 69			------
 70			`ValueError` if `other` is an invalid GPath
 71
 72			Examples
 73			--------
 74			```python
 75			GPath("/")
 76			GPath("/usr/bin")
 77			GPath("C:/Program Files")
 78			```
 79		"""
 80		self._parts: tuple[str, ...] = tuple()  # root- or parent- relative path
 81		self._device: str = ""
 82		self._absolute: bool = False
 83		self._parent: int = 0
 84		if path is not None and path != "":
 85			if isinstance(path, GPath):
 86				path._validate()
 87				self._parts = path._parts
 88				self._device = path._device
 89				self._absolute = path._absolute
 90				self._parent = path._parent
 91			else:
 92				# Remove redundant '.'s and '..'s and use OS-default path separators
 93				path = os.path.normpath(path)  # sets empty path to '.' and removes trailing slash
 94
 95				if path == os.curdir:
 96					path = ""
 97				(self._device, path) = os.path.splitdrive(path)
 98				self._absolute = os.path.isabs(path)
 99
100				parts = path.split(os.sep)  # os.path.normpath previously rewrote the path to use os.sep
101				if len(parts) > 0 and parts[0] == "":  # First element is '' if root
102					parts = parts[1:]
103				if len(parts) > 0 and parts[-1] == "":  # Last element is '' if there is a trailing slash, which should only happen when the path is exactly root ('/')
104					parts = parts[:-1]
105
106				dotdot = 0
107				while dotdot < len(parts) and parts[dotdot] == GPath.parent_dir:  # GPath.parent_dir == '..' usually
108					dotdot += 1
109				self._parts = tuple(parts[dotdot:])
110				self._parent = dotdot

123	@staticmethod
124	def find_common(path1: GPathLike, path2: GPathLike, allow_current: bool=True, allow_parents: bool=False) -> Optional[GPath]:
125		"""
126			Find the longest common base path shared by the two paths, or return None if no such path exists.
127
128			A common base path might not exist if one path is an absolute path while the other is a relative path, or if the two paths are in different filesystems (with different device names), or in other cases as controlled by the `allow_current` and `allow_parents` options.
129
130			Parameters
131			----------
132			`path1`, `path2`
133			: the paths to compare
134
135			`allow_current`
136			: whether two non-parent relative paths that do not share any components should be considered to have a common base path, namely the imaginary current working directory. For instance, `GPath.find_common("some/rel/path", "another/rel/path")` will return `GPath("")` if set to True, or return None if set to False.
137
138			`allow_parents`
139			: whether two relative paths that are relative to different levels of parent directories should be considered to have a common base path, which is the highest level of parent directory between the two paths. For instance, `GPath.find_common("../rel/to/parent", "../../rel/to/grandparent")` will return `GPath("../..")` if set to True, or return None if set to False. **Warning**: when set to True, given a higher level of parent directory as output, it may not be possible to find the relative path to one of the inputs (see `relpath_from()`); in most cases False is more appropriate.
140
141			Returns
142			-------
143			`GPath`
144			: the longest common base path, which may be empty, if it exists
145
146			`None`
147			: otherwise
148
149			Raises
150			------
151			`ValueError` if either GPath is invalid
152
153			Examples
154			--------
155			```python
156			GPath.find_common("/usr/bin", "/usr/local/bin")               # GPath("/usr")
157			GPath.find_common("C:/Windows/System32", "C:/Program Files")  # GPath("C:/")
158			GPath.find_common("../Documents", "../Pictures")              # GPath("..")
159			```
160		"""
161		if isinstance(path1, GPath):
162			path1._validate()
163		else:
164			path1 = GPath(path1)
165		if isinstance(path2, GPath):
166			path2._validate()
167		else:
168			path2 = GPath(path2)
169
170		if path1._device != path2._device:
171			return None
172		if path1._absolute != path2._absolute:
173			return None
174
175		if allow_parents:
176			allow_current = True
177
178		parts = []
179		if path1._absolute:
180			common_path = GPath()
181			for part1, part2 in zip(path1._parts, path2._parts):
182				if part1 == part2:
183					parts.append(part1)
184			common_path._absolute = True
185			# dotdot must be 0
186		else:
187			if path1._parent != path2._parent:
188				if not allow_parents:
189					return None
190
191				common_path = GPath()
192				common_path._parent = max(path1._parent, path2._parent)
193			else:
194				common_path = GPath()
195				common_path._parent = path1._parent
196				for part1, part2 in zip(path1._parts, path2._parts):
197					if part1 == part2:
198						parts.append(part1)
199
200		common_path._device = path1._device
201		common_path._parts = tuple(parts)
202
203		if not allow_current and not bool(common_path):
204			if common_path != path1 or common_path != path2:
205				return None
206		return common_path

208	@staticmethod
209	def partition(*paths: Union[Collection[GPathLike], GPathLike], allow_current: bool=True, allow_parents: bool=False) -> dict[GPath, list[GPath]]:
210		"""
211			Partition a collection of paths based on shared common base paths such that each path belongs to one partition.
212
213			For each partition, return a list of relative paths from the base path of that partition to each corresponding input path within that partition, unless `allow_parents` is True (see below). If the input collection is ordered, the output order is preserved within each partition. If the input collection contains duplicates, the corresponding output lists will as well.
214
215			The number of partitions is minimised by merging partitions as much as possible, so that each partition represents the highest possible level base path. Two partitions can no longer be merged when there is no common base path between them, as determined by `GPath.find_common()`. This method takes the same optional arguments as `GPath.find_common()`, with the same default values.
216
217			Parameters
218			----------
219			`paths: Collection[GPath | str | os.PathLike]` or `*paths: GPath | str | os.PathLike`
220			: the paths to be partitioned, which can be given as either a list-like object or as variadic arguments
221
222			`allow_current`
223			: whether non-parent relative paths with no shared components should be considered to have a common base path (see `GPath.find_common()`)
224
225			`allow_parents`
226			: whether paths that are relative to different levels of parent directories should be considered to have a common base path (see `GPath.find_common()`). **Warning**: when set to True, the output lists for each partition are invalidated, and explicitly set to empty. This is because it is not possible in general to obtain a relative path from the base path to its members if the base path is a parent directory of a higher level than the member (see `relpath_from()`). This  option should be True if and only if the list of members in each partition are not of interest; in most cases False is more appropriate.
227
228			Returns
229			-------
230			a dictionary that maps the common base path of each partition to a list of relative paths
231
232			Raises
233			------
234			  `ValueError`
235			  if any of the GPaths are invalid
236
237			Examples
238			--------
239			```python
240			GPath.partition("/usr/bin", "/usr/local/bin", "../../doc", "C:/Windows", "C:/Program Files")
241
242			assert partitions == {
243				GPath("/usr")      : [GPath("bin"), GPath("local")],
244				GPath("../../doc") : [GPath("")],
245				GPath("C:/")       : [GPath("Windows"), GPath("Program Files")],
246			}
247			```
248		"""
249		flattened_paths: list[GPathLike] = []
250		for path_or_list in paths:
251			if _is_gpathlike(path_or_list):
252				flattened_paths.append(path_or_list)
253			else:
254				flattened_paths.extend(path_or_list)
255		gpaths = [path if isinstance(path, GPath) else GPath(path) for path in flattened_paths]
256
257		partition_map = {}
258		if len(gpaths) > 0:
259			if allow_parents == True:
260				partition_map[gpaths[0]] = []
261			else:
262				partition_map[gpaths[0]] = [gpaths[0]]
263
264		for path in gpaths[1:]:
265			partition_found = False
266			for partition in partition_map:
267				candidate_common = GPath.find_common(partition, path, allow_current=allow_current, allow_parents=allow_parents)
268				if candidate_common is not None:
269					partition_found = True
270					if candidate_common != partition:
271						partition_map[candidate_common] = partition_map[partition]
272						del partition_map[partition]
273					if allow_parents == False:
274						partition_map[candidate_common].append(path)
275					break
276			if not partition_found:
277				if allow_parents == True:
278					partition_map[path] = []
279				else:
280					partition_map[path] = [path]
281
282		for partition, path_list in partition_map.items():
283			partition_map[partition] = [path.subpath_from(partition) for path in path_list]
284
285		return partition_map

290		"""
291			Join a sequence of paths into a single path. Apart from the first item in the sequence, all subsequent paths should be relative paths and any absolute paths will be ignored.
292
293			Parameters
294			----------
295			`paths`: `Collection[GPath | str | os.PathLike]` or `*paths: GPath | str | os.PathLike`
296			: the paths to be combined, which can be given as either a list-like object or as variadic arguments
297
298			Returns
299			-------
300			the combined path
301
302			Raises
303			------
304			`ValueError` if any of the GPaths are invalid
305
306			Examples
307			--------
308			```python
309			GPath.join("usr", "local", "bin")          # GPath("usr/local/bin")
310			GPath.join("/usr/local/bin", "../../bin")  # GPath("/usr/bin")
311			GPath.join("C:/", "Windows")               # GPath("C:/Windows")
312			```
313		"""
314		flattened_paths: list[GPathLike] = []
315		for path_or_list in paths:
316			if _is_gpathlike(path_or_list):
317				flattened_paths.append(path_or_list)
318			else:
319				flattened_paths.extend(path_or_list)
320
321		if len(flattened_paths) == 0:
322			return GPath()
323
324		combined_path = flattened_paths[0]
325		if not isinstance(combined_path, GPath):
326			combined_path = GPath(combined_path)
327		for path in flattened_paths[1:]:
328			combined_path = combined_path + path
329
330		return combined_path

333	def get_parts(self, root: bool=True, parent: bool=True) -> list[str]:
334		"""
335			Convert the path to a list of strings that unambiguously represents the given path.
336
337			The list of strings is given in such a way that a valid path can be reconstructed as a single string using <code>GPath.separator.join(<var>g</var>.get_parts())</code>. By default, the list of strings represents the full input path, but this behaviour can be changed using the optional arguments.
338
339			If it is an absolute path, the first item in the returned list will contain the device name if it exists, or be an empty string otherwise. If the path is the filesystem root exactly, the returned list will contain exactly two items, with the second being an empty string.
340
341			If the path is relative to a parent directory, e.g. `../..`, each parent level will be given as a separate item in the returned list.
342
343			Parameters
344			----------
345			`root`
346			: whether to include any components that indicate the filesystem root or device name. If set to True, the returned list will represent an absolute path as described above if the input path is absolute. If set to False, the returned list will always represent a relative path.
347
348			`parent`
349			: whether to include any components that indicate parent directories for a relative path. If set to False, the returned list will always be either an absolute path or a path relative to the imaginary current working directory.
350
351			Returns
352			-------
353			list of strings representing the path
354
355			Examples
356			--------
357			```python
358			GPath("usr/local/bin").get_parts()     # ["usr", "local", "bin"]
359			GPath("/usr/bin").get_parts()          # ["", "usr", "bin"]
360			GPath("C:/Program Files").get_parts()  # ["C:", "Program Files"]
361			GPath("../../Documents").get_parts()   # ["..", "..", "Documents"]
362			```
363		"""
364		if root and self._absolute:
365			if len(self._parts) == 0:
366				return [self._device, ""]
367
368			base_parts = [self._device]
369
370		elif parent and self._parent > 0:
371			base_parts = self.get_parent_parts()
372
373		else:
374			if len(self._parts) == 0:
375				# bool(self) == False
376				return [GPath.current_dir]
377
378			base_parts = []
379
380		return base_parts + list(self._parts)

457	def subpath_from(self, base: GPathLike) -> Optional[GPath]:
458		"""
459			Find the relative subpath from `base` to `self` if possible and if `base` contains `self`, or return None otherwise.
460
461			None will also be returned if there are unknown components in the subpath from `base` to `self`. For instance, if `self` is relative to the parent directory while `base` is relative to the grandparent directory, the path from the grandparent directory `../..` to the parent directory `..` cannot be known.
462
463			Similar to `relpath_from()`, but `self` must be a descendent of `base`.
464
465			Parameters
466			----------
467			`base`
468			: the base path that the relative subpath should start from
469
470			Returns
471			-------
472			`GPath`
473			: relative subpath from `base` to `self`, which may be empty, if it exists
474
475			`None`
476			: otherwise
477
478			Raises
479			------
480			`ValueError` if either `self` or `base` is an invalid GPath
481
482			Examples
483			--------
484			```python
485			GPath("/usr/local/bin").subpath_from("/usr")      # GPath("local/bin")
486			GPath("/usr/bin").subpath_from("/usr/local/bin")  # None
487			GPath("/usr/bin").subpath_from("../Documents")    # None
488			```
489		"""
490		if not isinstance(base, GPath):
491			base = GPath(base)
492
493		if GPath.find_common(self, base, allow_current=True, allow_parents=False) is not None and self in base:
494			# If self._dotdot > base._dotdot, self is not in base, whereas if self._dotdot < base._dotdot, path from base to self's parent cannot be known
495			base_length = len(base._parts)
496			new_path = GPath()
497			new_path._parts = self._parts[base_length:]  # () when self == base
498			return new_path
499		else:
500			return None

503	def relpath_from(self, origin: GPathLike) -> Optional[GPath]:
504		"""
505			Find the relative path from `origin` to `self` if possible, or return None otherwise.
506
507			None will also be returned if there are unknown components in the relative path from `origin` to `self`. For instance, if `self` is relative to the parent directory while `base` base is relative to the grandparent directory, the path from the grandparent directory `../..` to the parent directory `..` cannot be known.
508
509			Similar to `subpath_from()`, but `self` does not need to be a descendent of `origin`.
510
511			Parameters
512			----------
513			`origin`
514			: the origin that the relative path should start from
515
516			Returns
517			-------
518			`GPath`
519			: relative path from `origin` to `self`, which may be empty, if it exists
520
521			`None`
522			: otherwise
523
524			Raises
525			------
526			`ValueError` if either `self` or `origin` is an invalid GPath
527
528			Examples
529			--------
530			```python
531			GPath("/usr/local/bin").subpath_from("/usr")      # GPath("local/bin")
532			GPath("/usr/bin").subpath_from("/usr/local/bin")  # GPath("../../bin")
533			GPath("/usr/bin").subpath_from("../Documents")    # None
534			```
535		"""
536		self._validate()
537		if not isinstance(origin, GPath):
538			origin = GPath(origin)
539
540		if origin._absolute:
541			common = GPath.find_common(self, origin)
542			if common is None:
543				return None
544
545			new_path = GPath()
546			new_path._parent = len(origin) - len(common)
547			new_path._parts = self._parts[len(common):]
548			return new_path
549
550		else:
551			common = GPath.find_common(self, origin, allow_current=True, allow_parents=True)
552			if common is None:
553				return None
554			if common._parent > self._parent:
555				return None  # Path from common to self's parent cannot be known
556
557			# common._dotdot == self._dotdot
558			# origin._dotdot <= self._dotdot
559
560			new_path = GPath()
561			if len(common) == 0:
562				if origin._parent == self._parent:
563					new_path._parent = len(origin)
564				else:
565					new_path._parent = (common._parent - origin._parent) + len(origin)
566				new_path._parts = self._parts
567			else:
568				new_path._parent = len(origin) - len(common)
569				new_path._parts = self._parts[len(common):]
570
571			return new_path

621			GPath("/") < GPath("/usr")     # True
622			```
623		"""
624		if not isinstance(other, GPath):
625			other = GPath(other)
626		return ((not self._absolute, self._device, -1 * self._parent) + self._parts) < ((not other._absolute, other._device, -1 * other._parent) + other._parts)

740	def __add__(self, other: GPathLike) -> GPath:
741		"""
742			Add (append) `other` to the end of `self` if `other` is a relative path, and return a new copy.
743
744			If `other` is an absolute path, or if `other` has a different device name, add nothing and return a copy of `self`.
745
746			Usage: <code><var>self</var> + <var>other</var></code>
747
748			Raises `ValueError` if either GPath is invalid
749
750			Examples
751			--------
752			```python
753			GPath("/usr") + GPath("local/bin)                            # GPath("/usr/local/bin")
754			GPath("C:/Windows/System32") + GPath("../SysWOW64/drivers")  # GPath("C:/Windows/SysWOW64/drivers")
755			GPath("..") + GPath("../..")                                 # GPath("../../..")
756			```
757		"""
758		if isinstance(other, GPath):
759			other._validate
760		else:
761			other = GPath(other)
762
763		if other._absolute:
764			return GPath(self)
765		elif other._device != None and other._device != "" and self._device != other._device:
766			return GPath(self)
767		else:
768			new_path = GPath(self)
769			new_parts = [part for part in self._parts]
770			for i in range(other._parent):
771				if len(new_parts) > 0:
772					new_parts.pop()
773				elif not new_path._absolute:
774					new_path._parent += 1
775				else:
776					pass  # parent of directory of root is still root
777
778			new_parts.extend(other._parts)
779			new_path._parts = tuple(new_parts)
780			return new_path

784		"""
785			Remove `n` components from the end of the path and return a new copy.
786
787			Usage: <code><var>self</var> - <var>n</var></code>

789			Raises `ValueError` if `self` is an invalid GPath or if `n` is negative

797			GPath("/") - 1                    # GPath("/")
798			```
799		"""

801			raise ValueError("cannot subtract a negative number of components from the path; use __add__() instead")

811				pass  # removing components from root should still give root

817		"""
818			Duplicate the named components of `self` `n` times and return a new path with the duplicated components.
819
820			Named components will be duplicated separately from the components representing a parent directory. If `self` is an absolute path, only the relative components will be duplicated.
821
822			If `n` is 0, the result is an empty path (either relative or absolute).
823
824			Usage: <code><var>self</var> * <var>n</var></code>
825
826			Raises `ValueError` if `self` is an invalid GPath or if `n` is negative.
827
828			Examples
829			--------

831			GPath("/usr/bin") * 2    # GPath("/usr/bin/usr/bin")

834			```
835		"""

837			raise ValueError("cannot multiply path by a negative integer")
838		new_path = GPath(self)
839		new_path._parent = self._parent * n
840		new_path._parts = self._parts * n

844	def __lshift__(self, n: int) -> GPath:
845		"""
846			Move the imaginary current working directory `n` steps up the filesystem tree.
847
848			If `self` is a relative path, remove up to `n` levels of parent directories from the start of the path and return a copy. If it is an absolute path, return a copy of `self` unchanged.
849
850			If `n` is negative, this is equivalent to `__rshift__(-n)`.
851
852			Usage: <code><var>self</var> << <var>n</var></code>
853
854			Raises `ValueError` if `self` is an invalid GPath.
855
856			Examples
857			--------
858			```python
859			GPath("../SysWOW64/drivers") << 1  # GPath("SysWOW64/drivers")
860			GPath("../doc") << 2               # GPath("doc")
861			GPath("/usr/bin") << 2             # GPath("/usr/bin")
862			```
863		"""
864		if n < 0:
865			return self.__rshift__(-1 * n)
866		new_path = GPath(self)
867		if not new_path._absolute:
868			new_path._parent = max(new_path._parent - n, 0)
869		return new_path

873		"""
874			Move the imaginary current working directory `n` steps down the filesystem tree.
875
876			If `self` is a relative path, add `n` levels of parent directories to the start of the path and return a copy. If it is an absolute path, return a copy of `self` unchanged.
877
878			If `n` is negative, this is equivalent to `__lshift__(-n)`.
879
880			Usage: <code><var>self</var> >> <var>n</var></code>
881
882			Raises `ValueError` if `self` is an invalid GPath
883
884			Examples
885			--------
886			```python
887			GPath("../SysWOW64/drivers") >> 1  # GPath("../../SysWOW64/drivers")
888			GPath("/usr/bin") >> 2             # GPath("/usr/bin")
889			```
890		"""
891		if n < 0:
892			return self.__lshift__(-1 * n)
893		new_path = GPath(self)
894		if not new_path._absolute:
895			new_path._parent += n
896		return new_path

899		# Check if self is in a valid state
900		if self._parent < 0:
901			raise ValueError(f"invalid GPath, _parent cannot be negative: {repr(self)}")
902		if self._absolute:

 60	def __init__(self, path: Union[PathLike, GPath, None]="") -> None:
 61		"""
 62			Initialise a normalised and generalised abstract file path, possibly by copying an existing GPath object.
 63
 64			Parameters
 65			----------
 66			`path`
 67			: path-like object representing a (possibly unnormalised) file path, or a GPath object to be copied
 68
 69			Raises
 70			------
 71			`ValueError` if `other` is an invalid GPath
 72
 73			Examples
 74			--------
 75			```python
 76			GPath("/")
 77			GPath("/usr/bin")
 78			GPath("C:/Program Files")
 79			```
 80		"""
 81		self._parts: tuple[str, ...] = tuple()  # root- or parent- relative path
 82		self._device: str = ""
 83		self._absolute: bool = False
 84		self._parent: int = 0
 85		if path is not None and path != "":
 86			if isinstance(path, GPath):
 87				path._validate()
 88				self._parts = path._parts
 89				self._device = path._device
 90				self._absolute = path._absolute
 91				self._parent = path._parent
 92			else:
 93				# Remove redundant '.'s and '..'s and use OS-default path separators
 94				path = os.path.normpath(path)  # sets empty path to '.' and removes trailing slash
 95
 96				if path == os.curdir:
 97					path = ""
 98				(self._device, path) = os.path.splitdrive(path)
 99				self._absolute = os.path.isabs(path)
100
101				parts = path.split(os.sep)  # os.path.normpath previously rewrote the path to use os.sep
102				if len(parts) > 0 and parts[0] == "":  # First element is '' if root
103					parts = parts[1:]
104				if len(parts) > 0 and parts[-1] == "":  # Last element is '' if there is a trailing slash, which should only happen when the path is exactly root ('/')
105					parts = parts[:-1]
106
107				dotdot = 0
108				while dotdot < len(parts) and parts[dotdot] == GPath.parent_dir:  # GPath.parent_dir == '..' usually
109					dotdot += 1
110				self._parts = tuple(parts[dotdot:])
111				self._parent = dotdot

124	@staticmethod
125	def find_common(path1: GPathLike, path2: GPathLike, allow_current: bool=True, allow_parents: bool=False) -> Optional[GPath]:
126		"""
127			Find the longest common base path shared by the two paths, or return None if no such path exists.
128
129			A common base path might not exist if one path is an absolute path while the other is a relative path, or if the two paths are in different filesystems (with different device names), or in other cases as controlled by the `allow_current` and `allow_parents` options.
130
131			Parameters
132			----------
133			`path1`, `path2`
134			: the paths to compare
135
136			`allow_current`
137			: whether two non-parent relative paths that do not share any components should be considered to have a common base path, namely the imaginary current working directory. For instance, `GPath.find_common("some/rel/path", "another/rel/path")` will return `GPath("")` if set to True, or return None if set to False.
138
139			`allow_parents`
140			: whether two relative paths that are relative to different levels of parent directories should be considered to have a common base path, which is the highest level of parent directory between the two paths. For instance, `GPath.find_common("../rel/to/parent", "../../rel/to/grandparent")` will return `GPath("../..")` if set to True, or return None if set to False. **Warning**: when set to True, given a higher level of parent directory as output, it may not be possible to find the relative path to one of the inputs (see `relpath_from()`); in most cases False is more appropriate.
141
142			Returns
143			-------
144			`GPath`
145			: the longest common base path, which may be empty, if it exists
146
147			`None`
148			: otherwise
149
150			Raises
151			------
152			`ValueError` if either GPath is invalid
153
154			Examples
155			--------
156			```python
157			GPath.find_common("/usr/bin", "/usr/local/bin")               # GPath("/usr")
158			GPath.find_common("C:/Windows/System32", "C:/Program Files")  # GPath("C:/")
159			GPath.find_common("../Documents", "../Pictures")              # GPath("..")
160			```
161		"""
162		if isinstance(path1, GPath):
163			path1._validate()
164		else:
165			path1 = GPath(path1)
166		if isinstance(path2, GPath):
167			path2._validate()
168		else:
169			path2 = GPath(path2)
170
171		if path1._device != path2._device:
172			return None
173		if path1._absolute != path2._absolute:
174			return None
175
176		if allow_parents:
177			allow_current = True
178
179		parts = []
180		if path1._absolute:
181			common_path = GPath()
182			for part1, part2 in zip(path1._parts, path2._parts):
183				if part1 == part2:
184					parts.append(part1)
185			common_path._absolute = True
186			# dotdot must be 0
187		else:
188			if path1._parent != path2._parent:
189				if not allow_parents:
190					return None
191
192				common_path = GPath()
193				common_path._parent = max(path1._parent, path2._parent)
194			else:
195				common_path = GPath()
196				common_path._parent = path1._parent
197				for part1, part2 in zip(path1._parts, path2._parts):
198					if part1 == part2:
199						parts.append(part1)
200
201		common_path._device = path1._device
202		common_path._parts = tuple(parts)
203
204		if not allow_current and not bool(common_path):
205			if common_path != path1 or common_path != path2:
206				return None
207		return common_path

209	@staticmethod
210	def partition(*paths: Union[Collection[GPathLike], GPathLike], allow_current: bool=True, allow_parents: bool=False) -> dict[GPath, list[GPath]]:
211		"""
212			Partition a collection of paths based on shared common base paths such that each path belongs to one partition.
213
214			For each partition, return a list of relative paths from the base path of that partition to each corresponding input path within that partition, unless `allow_parents` is True (see below). If the input collection is ordered, the output order is preserved within each partition. If the input collection contains duplicates, the corresponding output lists will as well.
215
216			The number of partitions is minimised by merging partitions as much as possible, so that each partition represents the highest possible level base path. Two partitions can no longer be merged when there is no common base path between them, as determined by `GPath.find_common()`. This method takes the same optional arguments as `GPath.find_common()`, with the same default values.
217
218			Parameters
219			----------
220			`paths: Collection[GPath | str | os.PathLike]` or `*paths: GPath | str | os.PathLike`
221			: the paths to be partitioned, which can be given as either a list-like object or as variadic arguments
222
223			`allow_current`
224			: whether non-parent relative paths with no shared components should be considered to have a common base path (see `GPath.find_common()`)
225
226			`allow_parents`
227			: whether paths that are relative to different levels of parent directories should be considered to have a common base path (see `GPath.find_common()`). **Warning**: when set to True, the output lists for each partition are invalidated, and explicitly set to empty. This is because it is not possible in general to obtain a relative path from the base path to its members if the base path is a parent directory of a higher level than the member (see `relpath_from()`). This  option should be True if and only if the list of members in each partition are not of interest; in most cases False is more appropriate.
228
229			Returns
230			-------
231			a dictionary that maps the common base path of each partition to a list of relative paths
232
233			Raises
234			------
235			  `ValueError`
236			  if any of the GPaths are invalid
237
238			Examples
239			--------
240			```python
241			GPath.partition("/usr/bin", "/usr/local/bin", "../../doc", "C:/Windows", "C:/Program Files")
242
243			assert partitions == {
244				GPath("/usr")      : [GPath("bin"), GPath("local")],
245				GPath("../../doc") : [GPath("")],
246				GPath("C:/")       : [GPath("Windows"), GPath("Program Files")],
247			}
248			```
249		"""
250		flattened_paths: list[GPathLike] = []
251		for path_or_list in paths:
252			if _is_gpathlike(path_or_list):
253				flattened_paths.append(path_or_list)
254			else:
255				flattened_paths.extend(path_or_list)
256		gpaths = [path if isinstance(path, GPath) else GPath(path) for path in flattened_paths]
257
258		partition_map = {}
259		if len(gpaths) > 0:
260			if allow_parents == True:
261				partition_map[gpaths[0]] = []
262			else:
263				partition_map[gpaths[0]] = [gpaths[0]]
264
265		for path in gpaths[1:]:
266			partition_found = False
267			for partition in partition_map:
268				candidate_common = GPath.find_common(partition, path, allow_current=allow_current, allow_parents=allow_parents)
269				if candidate_common is not None:
270					partition_found = True
271					if candidate_common != partition:
272						partition_map[candidate_common] = partition_map[partition]
273						del partition_map[partition]
274					if allow_parents == False:
275						partition_map[candidate_common].append(path)
276					break
277			if not partition_found:
278				if allow_parents == True:
279					partition_map[path] = []
280				else:
281					partition_map[path] = [path]
282
283		for partition, path_list in partition_map.items():
284			partition_map[partition] = [path.subpath_from(partition) for path in path_list]
285
286		return partition_map

505		"""
506			Find the relative path from `origin` to `self` if possible, or return None otherwise.
507
508			None will also be returned if there are unknown components in the relative path from `origin` to `self`. 509
510			Similar to `subpath_from()`, but `self` does not need to be a descendent of `origin`.
511
512			Parameters
513			----------
514			`origin`
515			: the origin that the relative path should start from
516
517			Returns
518			-------
519			`GPath`
520			: relative path from `origin` to `self`, which may be empty, if it exists
521
522			`None`
523			: otherwise

526			------
527			`ValueError` if either `self` or `origin` is an invalid GPath

530			--------
531			```python

534			GPath("/usr/bin").subpath_from("../Documents")    # None

538		if not isinstance(origin, GPath):
539			origin = GPath(origin)
540
541		if origin._absolute:
542			common = GPath.find_common(self, origin)
543			if common is None:

547			new_path._parent = len(origin) - len(common)
548			new_path._parts = self._parts[len(common):]

568			else:
